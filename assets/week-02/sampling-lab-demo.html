Stratified sampling

teams <- c(rep("yankees", 40), 
           rep("padres", 40), 
           rep("mariners", 40), 
           rep("dodgers", 40))
salary <- c(rnorm(40, mean = 25), 
            rnorm(40, 16), 
            rnorm(40, 23), 
            rnorm(40, 15))
df <- data.frame(teams, salary)
head(df)

Population view

SRS vs Stratified

# population mean
mean(df$salary)
## [1] 19.81544
# SRS
df %>% sample_n(40) %>% summarize(mean(salary))
##   mean(salary)
## 1     19.05474
# Stratified sample
df %>% 
  group_by(teams) %>% 
  sample_n(10) %>% 
  ungroup() %>%
  summarize(mean(salary))
## # A tibble: 1 x 1
##   `mean(salary)`
##            <dbl>
## 1       20.01825
